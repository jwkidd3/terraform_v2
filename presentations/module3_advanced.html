Standard Module Organization
# Enterprise Security Policy Suite
import "tfplan/v2" as tfplan
import "strings"
import "types"

# SOC 2 Compliance - Resource Tagging Requirements
required_tags = [
    "Environment",
    "Owner", 
    "CostCenter",
    "DataClassification",
    "ComplianceFramework",
    "BackupTier",
    "BusinessUnit"
]

# PCI DSS - Network Security Controls
allowed_instance_types = [
    "t3.medium", "t3.large", "t3.xlarge",
    "c5.large", "c5.xlarge", "c5.2xlarge",
    "m5.large", "m5.xlarge", "m5.2xlarge"
]

# HIPAA - Encryption Requirements
encryption_required_resources = [
    "aws_ebs_volume",
    "aws_rds_instance", 
    "aws_s3_bucket",
    "aws_redshift_cluster",
    "aws_kinesis_stream"
]

# Main policy rule - Tag compliance
mandatory_tags = rule {
    all tfplan.resource_changes as _, changes {
        changes.mode is "managed"
        changes.type in ["aws_instance", "aws_rds_instance", "aws_s3_bucket"]
        
        # Check all required tags exist
        all required_tags as tag {
            changes.change.after.tags[tag] is not null
        }
        
        # Validate tag values are not empty
        all required_tags as tag {
            length(changes.change.after.tags[tag]) > 0
        }
        
        # Environment tag must be valid
        changes.change.after.tags["Environment"] in ["development", "staging", "production"]
        
        # Data classification must be specified
        changes.change.after.tags["DataClassification"] in ["public", "internal", "confidential", "restricted"]
    }
}

# Security policy - Instance restrictions
approved_instance_types = rule {
    all tfplan.resource_changes as _, changes {
        changes.type is "aws_instance"
        changes.change.after.instance_type in allowed_instance_types
        
        # Production instances must be encrypted
        changes.change.after.tags["Environment"] is "production" implies
            changes.change.after.ebs_block_device else [] as devices {
                all devices as device {
                    device.encrypted is true
                }
            }
    }
}

# Network security - No overly permissive security groups
security_group_compliance = rule {
    all tfplan.resource_changes as _, changes {
        changes.type is "aws_security_group"
        
        # No SSH from anywhere
        all changes.change.after.ingress as rule {
            not (rule.from_port <= 22 and 
                 rule.to_port >= 22 and 
                 "0.0.0.0/0" in rule.cidr_blocks)
        }
        
        # No RDP from anywhere  
        all changes.change.after.ingress as rule {
            not (rule.from_port <= 3389 and 
                 rule.to_port >= 3389 and 
                 "0.0.0.0/0" in rule.cidr_blocks)
        }
    }
}

# Cost control - Prevent expensive resources
cost_control = rule {
    all tfplan.resource_changes as _, changes {
        changes.type is "aws_instance"
        
        # Expensive instances require approval tag
        changes.change.after.instance_type matches "^(c5|m5|r5)\\.(4|8|12|16|24)xlarge$" implies
            changes.change.after.tags["ApprovedForHighCost"] is "true"
    }
}

# Encryption enforcement
encryption_compliance = rule {
    all tfplan.resource_changes as _, changes {
        changes.type in encryption_required_resources
        
        changes.type is "aws_ebs_volume" implies changes.change.after.encrypted is true
        changes.type is "aws_rds_instance" implies changes.change.after.storage_encrypted is true
        changes.type is "aws_s3_bucket" implies changes.change.after.server_side_encryption_configuration[0].rule[0].apply_server_side_encryption_by_default[0].sse_algorithm is not null
    }
}

# Main policy evaluation
main = rule {
    mandatory_tags and
    approved_instance_types and 
    security_group_compliance and
    cost_control and
    encryption_compliance
}

package terraform.cost

import future.keywords.in

# Define instance type costs (enterprise example)
instance_costs := {
    "t2.nano":   5,
    "t2.micro":  10,
    "t2.small":  20,
    "t3.medium": 40,
    "t3.large":  80,
    "t3.xlarge": 160
}

# Warn about expensive instances
warn[msg] {
    resource := input.planned_values.root_module.resources[_]
    resource.type == "aws_instance"
    
    cost := instance_costs[resource.values.instance_type]
    cost > 50
    
    msg := sprintf("Instance %s uses expensive type %s ($%d/month)", 
        [resource.address, resource.values.instance_type, cost])
}

# Deny deployment if total cost exceeds budget
budget_limit := 1000

deny[msg] {
    total_cost := sum([
        instance_costs[resource.values.instance_type] |
        resource := input.planned_values.root_module.resources[_]
        resource.type == "aws_instance"
    ])
    
    total_cost > budget_limit
    msg := sprintf("Total deployment cost $%d exceeds budget of $%d", 
        [total_cost, budget_limit])
}
